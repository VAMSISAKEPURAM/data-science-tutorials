üîé 5. Example: Cleaning Data with Python

# Import libraries
import pandas as pd

# Load dataset
df = pd.read_csv("data.csv")

# Fill missing values
df["Age"].fillna(df["Age"].mean(), inplace=True)

# Remove duplicates
df.drop_duplicates(inplace=True)

# Standardize text columns
df["Country"] = df["Country"].str.title()
